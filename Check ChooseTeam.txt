--// Blox Fruits - Kick nếu chưa chọn team trong 60s
repeat task.wait() until game:IsLoaded()
local player = game.Players.LocalPlayer
local TIMEOUT = 90
local choseTeam = false

-- GUI Setup
local screenGui = Instance.new("ScreenGui")
screenGui.Name = "TeamCheckUI"
screenGui.ResetOnSpawn = false
screenGui.Parent = player:WaitForChild("PlayerGui")

local frame = Instance.new("Frame")
frame.Size = UDim2.new(0, 300, 0, 120)
frame.Position = UDim2.new(0.5, -150, 0.2, 0)
frame.BackgroundColor3 = Color3.fromRGB(20, 20, 20)
frame.BackgroundTransparency = 0.2
frame.BorderSizePixel = 0
frame.Visible = false
frame.Parent = screenGui

local title = Instance.new("TextLabel")
title.Size = UDim2.new(1, 0, 0, 30)
title.BackgroundTransparency = 1
title.Font = Enum.Font.SourceSansBold
title.TextSize = 20
title.Text = "⚠️ Kiểm tra chọn team"
title.TextColor3 = Color3.fromRGB(255, 255, 0)
title.Parent = frame

local msg = Instance.new("TextLabel")
msg.Size = UDim2.new(1, -20, 0, 60)
msg.Position = UDim2.new(0, 10, 0, 35)
msg.BackgroundTransparency = 1
msg.Font = Enum.Font.SourceSans
msg.TextSize = 16
msg.TextWrapped = true
msg.TextColor3 = Color3.fromRGB(255, 255, 255)
msg.Text = "Bạn chưa chọn team. Hãy chọn Pirates hoặc Marines trước khi hết thời gian!"
msg.Parent = frame

local timeLabel = Instance.new("TextLabel")
timeLabel.Size = UDim2.new(1, 0, 0, 20)
timeLabel.Position = UDim2.new(0, 0, 1, -25)
timeLabel.BackgroundTransparency = 1
timeLabel.Font = Enum.Font.SourceSansBold
timeLabel.TextSize = 18
timeLabel.TextColor3 = Color3.fromRGB(255, 170, 0)
timeLabel.Text = ""
timeLabel.Parent = frame

-- Hàm check choose team
local function hasTeam()
    return player.Team ~= nil and tostring(player.Team) ~= "nil"
end

-- choose team = stop
if hasTeam() then
    print("[TeamCheck] Người chơi đã có team:", player.Team.Name)
    return
end

-- đếm ngược giây
frame.Visible = true
print("[TeamCheck] Bắt đầu kiểm tra chọn team trong 60s...")

local startTick = tick()

local conn
conn = player:GetPropertyChangedSignal("Team"):Connect(function()
    if hasTeam() then
        choseTeam = true
        local elapsed = math.floor(tick() - startTick)
        print(string.format("[TeamCheck] Đã chọn team %s sau %ds", player.Team.Name, elapsed))
        frame.Visible = false
        conn:Disconnect()
    end
end)

for i = TIMEOUT, 0, -1 do
    if choseTeam then break end
    timeLabel.Text = "⏳ Còn lại: " .. i .. "s"
    task.wait(1)
end

if not choseTeam then
    print("[TeamCheck] Không chọn team trong 60s → kick player.")
    frame.Visible = true
    title.Text = "Hết thời gian!"
    msg.Text = "No choose team"
    timeLabel.Text = ""
    task.wait(2)
    game.Players.LocalPlayer:Kick("Bạn không chọn team trong 60 giây.")
end
